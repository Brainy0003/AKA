/* FunctionAdd.cpp
   Demonstrates passing parameters to 
   a function for simple addition
*/
#include <iostream>
using namespace std ;       
/* ------ Function Addition ------ */
int Addition ( int num1 , int num2 )
{
         int sum ;
         sum = num1 + num2 ;
         return ( sum ) ;
} 
/* ------- Addition End -------- */
int main ( void )
{
         int My_Answer = Addition ( 1 , 2 ) ;
         cout << "The result is " << My_Answer ;
         return 0 ;
}

 

/* FunctionHello.cpp
   Hello World example using a Function
*/
         #include <iostream>
         using namespace std; 
void My_Function ( void )
         {
         cout << "I function, therefore I am";
         }
int main ( void )
         {
         My_Function ();
         return 0;
         }

In the above example we use the void function type, because the function takes no parameters and returns no value, it just prints something to the screen.
